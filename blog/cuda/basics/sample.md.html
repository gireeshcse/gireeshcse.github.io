__global__

It tells CUDA c++ compiler that this function runs on the GPU and can be called from CPU Code.

/**
* CUDA kernel function to add the elements of 2 arrays on the GPU
*/
__global__
void add(int n, float *x, float *y)
{
    for(int i=0;i<n;i++)
    {
        y[i] = x[i] + y[i];
    }
}
